improveddemocracy_decisions = {

	embargo_hostile_country = {
	
		cost = 10
		
		target_trigger = {
			FROM = {
				NOT = { has_government = democratic }
			}
		}
		
		visible = {
			FROM = {
				NOT = {
					has_government = democratic
					is_in_faction = yes
				}
				OR = {
					has_offensive_war = yes
					has_added_tension_amount > 30
				}
			}
		}
		
		available = {
			has_political_power > 10
			NOT = {
				FROM = {
					has_opinion_modifier = embargo
				}
			}
			NOT = {
				has_country_flag = is_giving_embargo
			}
		}
		
		days_remove = 180

		
		complete_effect = {
			add_opinion_modifier = {
				target = FROM
				modifier = embargo
			}
			reverse_add_opinion_modifier = {
				target = FROM
				modifier = gave_embargo
			}
			FROM = {
				save_event_target_as = embargo_receiver
				country_event = {
					id = improveddemocracy.1
					days = 1
				}
			}
			ROOT = {
				save_event_target_as = embargo_giver
			}
			add_country_flag = is_giving_embargo
		}
		
		remove_effect = {
			remove_opinion_modifier = {
				target = FROM
				modifier = embargo
			}
			FROM = {
				remove_opinion_modifier = {
					target = FROM
					modifier = gave_embargo
				}
			}
			country_event = {
				id = improveddemocracy.6
				days = 1
			}
			clr_country_flag = is_giving_embargo
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				ROOT = {
					has_opinion = {
						target = FROM
						value > 50
					}			
				}
			}
			modifier = {
				factor = 0.1
				ROOT = {
					has_opinion = {
						target = FROM
						value > 10
					}			
				}
			}
		}
		
	}
	
}
